// ============================================================ // 🐛 修复 1：Date 对象序列化丢失 // 问题：Date 对象存入 localStorage 后变成 ISO 字符串， // 刷新后 date >= cutoff 比较永远失败 // 方案：parseXlsx 解析时统一将 Date 转为 "yyyy-mm-dd" 字符串 // ============================================================ /** * 通用日期解析：支持 Date对象 / ISO字符串 / yyyy-mm-dd / 各种分隔符 * 始终返回 Date 对象或 null */ function parseDate(v) { if (v == null || v === '') return null; // 如果已经是 Date 对象 if (v instanceof Date) { return isNaN(v.getTime()) ? null : v; } if (typeof v === 'number') { // Excel 序列号日期（天数从 1900-01-01 起） if (v > 30000 && v < 60000) { const d = new Date(Math.round((v - 25569) * 86400 * 1000)); return isNaN(d.getTime()) ? null : d; } return null; } if (typeof v !== 'string') return null; let s = v.trim(); // ISO 格式 "2024-01-15T00:00:00.000Z" if (/^\d{4}-\d{2}-\d{2}(T|$)/.test(s)) { const d = new Date(s.substring(0, 10) + 'T00:00:00'); return isNaN(d.getTime()) ? null : d; } // "dd/mm/yyyy" 或 "mm/dd/yyyy" — 尝试多种格式 let m; if ((m = s.match(/^(\d{1,2})[\/\-\.](\d{1,2})[\/\-\.](\d{4})$/))) { // 尝试 mm/dd/yyyy let d = new Date(+m[3], +m[1] - 1, +m[2]); if (!isNaN(d.getTime())) return d; // 尝试 dd/mm/yyyy d = new Date(+m[3], +m[2] - 1, +m[1]); if (!isNaN(d.getTime())) return d; } if ((m = s.match(/^(\d{4})[\/\-\.](\d{1,2})[\/\-\.](\d{1,2})$/))) { const d = new Date(+m[1], +m[2] - 1, +m[3]); return isNaN(d.getTime()) ? null : d; } // 最后兜底 const d = new Date(s); return isNaN(d.getTime()) ? null : d; } /** * 将日期值标准化为 "yyyy-mm-dd" 字符串，确保 JSON 序列化安全 */ function dateToStr(v) { const d = (v instanceof Date) ? v : parseDate(v); * * 示例改法（找到原函数中类似代码）： * * function parseXlsx(file, callback) { * ... * const rows = XLSX.utils.sheet_to_json(ws, { defval: '' }); * const safeRows = sanitizeRowDates(rows); // ← 新增这一行 * ... // 后续用 safeRows 替代 rows * } */ // ============================================================ // 修复 dailyAvg 中的日期比较逻辑，兼容字符串日期 // ============================================================ /** * 修复后的日均销量计算 * 确保 cutoff 和 record.date 都通过 parseDate() 转为 Date 再比较 */ function dailyAvg(salesRecs, sku, days) { if (!salesRecs || salesRecs.length === 0) return 0; const now = new Date(); now.setHours(0, 0, 0, 0); const cutoff = new Date(now); cutoff.setDate(cutoff.getDate() - days); const skus = skuSet(sku); // 修复3中增强的版本 let totalQty = 0; let count = 0; for (let i = 0; i < salesRecs.length; i++) { const rec = salesRecs[i]; // 关键修复：始终通过 parseDate 将字符串还原为 Date 对象 const recDate = parseDate(rec.date); if (!recDate) continue; if (recDate >= cutoff && recDate <= now) { const recSku = String(rec.sku || '').trim().toUpperCase(); (totalQty / actualDays) : 0; } // ============================================================ // 🐛 修复 2：销量列名动态识别 // 问题：硬编码列名无法匹配 Shopify 不同导出格式 // 方案：动态检测列名，精确匹配 + 正则模糊匹配 + fallback // ============================================================ function procSales(rows) { if (!rows || rows.length === 0) { console.warn('[procSales] 无数据行'); String(r[nameCol] || '').trim() : ''; if (!dateVal || !skuVal) { skipped++; if (skipped <= 5) { console.log(`[procSales] 跳过行 ${i}:`, { dateVal, skuVal, raw: r }); } continue; } parsed++; results.push({ date: dateVal, // "yyyy-mm-dd" 字符串，安全存储 sku: skuVal.toUpperCase(), qty: qtyVal, name: nameVal }); } console.log(`[procSales] ✅ 解析完成: 成功 ${parsed} 行, 跳过 $ {skipped} 行, 总 ${rows.length} 行`); return results; } // ============================================================ // 🐛 修复 3：skuSet 反向映射缺失 // 问题：目标SKU无法反向找到同源的兄弟SKU // 方案：增加反向查找，若当前SKU是某映射的target， // 则将同一src下所有target也加入集合 // ============================================================ /** * 增强版 skuSet：正向 + 反向映射 * @param {string} sku - 查询的 SKU * @returns {Set} - 包含自身及所有关联 SKU 的集合（大写） * * 假设 skuMappings 结构为: * [ { src: "PF003-S+SA007-JP", targets: ["PF003-S+SA007-JP1", "PF003-S+SA007-JP2"], name: "..." }, ... ] */ function skuSet(sku) { const result = new Set(); if (!sku) return result; const upperSku = String(sku).trim().toUpperCase(); result.add(upperSku); // 读取映射表（根据你原来的变量名调整） const mappings = window.skuMappings || []; for (const mapping of mappings) { const src = String(mapping.src || '').trim().toUpperCase();
